### **Step 4 — Cron Job**

* Install `node-cron`: `npm i node-cron`
* Create `cron/cron.js`:

```js
const cron = require('node-cron');
const Product = require('../models/Product');
const getPrice = require('../utils/scraper');

function startCron() {
  // Every hour
  cron.schedule('0 * * * *', async () => {
    console.log('Running price update cron job...');
    const products = await Product.find();
    for (let product of products) {
      const newPrice = await getPrice(product.url);
      if (newPrice && newPrice !== product.currentPrice) {
        const oldPrice = product.currentPrice;
        const change = Math.round(((newPrice - oldPrice) / oldPrice) * 10) / 10;

        product.history.push({
          oldPrice,
          newPrice,
          change,
          date: new Date(),
        });

        product.currentPrice = newPrice;
        product.change = change;

        await product.save();
        console.log(`Updated ${product.name}: ${newPrice}`);
      }
    }
  });
}

module.exports = startCron;
```

* Call this cron in `server.js` after connecting DB:
